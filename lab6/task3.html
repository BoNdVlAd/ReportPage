Умова 3 завданяя

У звітному HTML-документі створити html-розміту, яка складається з наступних елементів : текст, кнопка, інпут (поле введення ) .

Кнопка "Приховати" ховає текст , виводячи зірочки замість введеної інформації, замінює назву кнопки на "Розкрити", при повторному натисканні текст знову стає доступним і кнопка набуває початкового вигляду. У звітному HTML-документі відобразити скрін програмного коду

Код 3 завданяя

let inputText = document.getElementById("inputText"); let toggleButton = document.getElementById("toggleButton"); let entity = ""; function toggleText() { if (inputText.value === "") { alert("Будь ласка, введіть текст"); return; } if (toggleButton.textContent === "Приховати") { entity = inputText.value inputText.value = inputText.value.replace(/./g, "*"); toggleButton.textContent = "Розкрити"; } else { inputText.value = entity; entity = ""; toggleButton.textContent = "Приховати"; } } toggleButton.addEventListener("click", toggleText);
